<#
=====================================================
ASP.NET CORE MVC / API CONTROLLER GENERATOR
=====================================================

Purpose:
- Generates Controllers from database_schema.json
- Uses Clean Architecture naming convention
- Applies DOC-BLOCK (XML-style) documentation
- Produces production-ready API structure

=====================================================
#>

# ==============================
# CONFIG
# ==============================
$schemaPath = "F:\DASHBOARD_APP\GIT_VERSION\TALLY\database_schema.json"
$basePath   = "F:\DASHBOARD_APP\GIT_VERSION\TALLY\TALLY_APP\Controllers"

# ==============================
# HELPERS
# ==============================
function Ensure-Folder {
    param([string]$path)

    if (!(Test-Path $path)) {
        New-Item -ItemType Directory -Path $path -Force | Out-Null
    }
}

# ==============================
# LOAD SCHEMA
# ==============================
$schema = Get-Content $schemaPath -Raw | ConvertFrom-Json

Write-Host "Controller generation started..."

# ==============================
# MAIN LOOP
# ==============================
foreach ($group in $schema.groups) {

    $groupName = $group.name -replace '\s',''

    foreach ($entity in $group.entities) {

        $entityName = $entity.name

        $modulePath = Join-Path $basePath $groupName
        Ensure-Folder $modulePath

        $filePath = Join-Path $modulePath "$entityName`Controller.cs"

        # ==============================
        # CONTROLLER CONTENT
        # ==============================
        $content = @"
using Microsoft.AspNetCore.Mvc;
using System.Collections.Generic;
using System.Threading.Tasks;
using $groupName.Services;
using $groupName.DTOs.Request;
using $groupName.DTOs.Response;

namespace POS_APP.Controllers.$groupName
{
    /// <summary>
    /// Controller for managing $entityName operations
    /// </summary>
    /// <remarks>
    /// This controller provides full CRUD operations for $entityName.
    /// It follows RESTful API standards and Clean Architecture principles.
    /// </remarks>
    [ApiController]
    [Route("api/$groupName/$entityName")]
    public class ${entityName}Controller : ControllerBase
    {
        private readonly I${entityName}Service _service;

        /// <summary>
        /// Initializes a new instance of the <see cref="$entityName Controller"/> class.
        /// </summary>
        /// <param name="service">Injected service layer for business logic</param>
        public ${entityName}Controller(I${entityName}Service service)
        {
            _service = service;
        }

        /// <summary>
        /// Retrieves all $entityName records
        /// </summary>
        /// <returns>List of $entityName response objects</returns>
        [HttpGet("all")]
        public async Task<ActionResult<List<${entityName}Response>>> GetAll()
        {
            return await _service.All();
        }

        /// <summary>
        /// Retrieves paginated list of $entityName records
        /// </summary>
        /// <returns>List of paginated $entityName data</returns>
        [HttpGet("index")]
        public async Task<ActionResult<List<${entityName}Response>>> Index()
        {
            return await _service.Index();
        }

        /// <summary>
        /// Retrieves a single $entityName by unique identifier
        /// </summary>
        /// <param name="id">Unique identifier of the record</param>
        /// <returns>Single $entityName record</returns>
        [HttpGet("view/{id}")]
        public async Task<ActionResult<${entityName}Response>> View(long id)
        {
            return await _service.View(id);
        }

        /// <summary>
        /// Creates a new $entityName record
        /// </summary>
        /// <param name="request">Request payload containing $entityName data</param>
        /// <returns>Created $entityName response</returns>
        [HttpPost("create")]
        public async Task<ActionResult<${entityName}Response>> Create(
            [FromBody] ${entityName}Request request)
        {
            return await _service.Create(request);
        }

        /// <summary>
        /// Updates an existing $entityName record
        /// </summary>
        /// <param name="id">Unique identifier of the record</param>
        /// <param name="request">Updated $entityName data</param>
        /// <returns>Updated $entityName response</returns>
        [HttpPut("update/{id}")]
        public async Task<ActionResult<${entityName}Response>> Update(
            long id,
            [FromBody] ${entityName}Request request)
        {
            return await _service.Update(id, request);
        }

        /// <summary>
        /// Deletes a $entityName record permanently
        /// </summary>
        /// <param name="id">Unique identifier of the record</param>
        /// <returns>Operation status message</returns>
        [HttpDelete("delete/{id}")]
        public async Task<IActionResult> Delete(long id)
        {
            await _service.Delete(id);
            return Ok(new { message = "Record deleted successfully" });
        }
    }
}
"@

        # ==============================
        # WRITE FILE
        # ==============================
        $content | Out-File -FilePath $filePath -Encoding UTF8

        Write-Host "Generated: $groupName -> $entityName"
    }
}

Write-Host "Generation completed successfully."
